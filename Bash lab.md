15. sort_du.sh

#!/bin/bash
dir="${1:-.}"
find "$dir" -type f -printf "%T+ %p\n" | sort | head -3 | while read -r date file; do
  modified=$(stat -c "%y" "$file" 2>/dev/null || echo "неизвестно")
  echo "$file — $modified"
done
